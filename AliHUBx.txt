local Players = game:GetService("Players")
local TweenService = game:GetService("TweenService")
local RunService = game:GetService("RunService")

local player = Players.LocalPlayer

-- Criar GUI
local gui = Instance.new("ScreenGui")
gui.Name = "TeleportGui"
gui.ResetOnSpawn = false
gui.Parent = player:WaitForChild("PlayerGui")

-- Criar Botão
local button = Instance.new("TextButton")
button.Name = "TeleportButton"
button.Size = UDim2.new(0, 150, 0, 50)
button.Position = UDim2.new(1, -160, 1, -60)
button.AnchorPoint = Vector2.new(0, 0)
button.BackgroundColor3 = Color3.new(0, 0, 0)
button.TextColor3 = Color3.new(1, 1, 1)
button.Font = Enum.Font.GothamBold
button.TextScaled = true
button.Text = "TELEPORTAR"
button.Parent = gui

-- Borda arredondada
local corner = Instance.new("UICorner")
corner.CornerRadius = UDim.new(0, 12)
corner.Parent = button

-- Borda RGB
local stroke = Instance.new("UIStroke")
stroke.ApplyStrokeMode = Enum.ApplyStrokeMode.Border
stroke.Thickness = 3
stroke.Transparency = 0
stroke.Parent = button

-- Cores RGB piscantes
local cores = {
	Color3.fromRGB(0, 170, 255),    -- Azul
	Color3.fromRGB(255, 255, 0),    -- Amarelo
	Color3.fromRGB(255, 0, 0),      -- Vermelho
	Color3.fromRGB(170, 0, 255),    -- Roxo
	Color3.fromRGB(150, 75, 0),     -- Marrom
}
local corIndex = 1

-- Loop de cor da borda
task.spawn(function()
	while true do
		stroke.Color = cores[corIndex]
		corIndex = corIndex % #cores + 1
		task.wait(0.25)
	end
end)

-- Função de teleportar e descer com Tween
local function teleportar()
	local character = player.Character or player.CharacterAdded:Wait()
	local root = character:WaitForChild("HumanoidRootPart")

	-- Teleportar pra cima
	local teleportHeight = 5000
	local startPos = root.Position
	root.CFrame = CFrame.new(startPos.X, teleportHeight, startPos.Z)

	-- Esperar 1 frame
	RunService.Heartbeat:Wait()

	-- Raycast para detectar a primeira parte embaixo
	local rayOrigin = root.Position
	local rayDirection = Vector3.new(0, -10000, 0)
	local params = RaycastParams.new()
	params.FilterDescendantsInstances = {character}
	params.FilterType = Enum.RaycastFilterType.Blacklist

	local result = workspace:Raycast(rayOrigin, rayDirection, params)
	if result then
		local targetY = result.Position.Y + 5
		local targetPos = Vector3.new(startPos.X, targetY, startPos.Z)

		local tween = TweenService:Create(
			root,
			TweenInfo.new(2, Enum.EasingStyle.Quad, Enum.EasingDirection.Out),
			{Position = targetPos}
		)
		tween:Play()
	end
end

-- Clique do botão
button.MouseButton1Click:Connect(teleportar)
