local Players = game:GetService("Players")
local RunService = game:GetService("RunService")

local player = Players.LocalPlayer

-- Criar GUI
local gui = Instance.new("ScreenGui")
gui.Name = "TeleportGui"
gui.ResetOnSpawn = false
gui.Parent = player:WaitForChild("PlayerGui")

-- Criar botão
local button = Instance.new("TextButton")
button.Name = "TeleportButton"
button.Size = UDim2.new(0, 150, 0, 50)
button.Position = UDim2.new(1, -160, 1, -60)
button.AnchorPoint = Vector2.new(0, 0)
button.BackgroundColor3 = Color3.new(0, 0, 0)
button.TextColor3 = Color3.new(1, 1, 1)
button.Font = Enum.Font.GothamBold
button.TextScaled = true
button.Text = "TELEPORTAR"
button.Draggable = true -- permite arrastar
button.Active = true -- necessário para Draggable
button.Selectable = true
button.Parent = gui

-- Borda arredondada
local corner = Instance.new("UICorner")
corner.CornerRadius = UDim.new(0, 12)
corner.Parent = button

-- UIStroke (borda RGB)
local stroke = Instance.new("UIStroke")
stroke.ApplyStrokeMode = Enum.ApplyStrokeMode.Border
stroke.Thickness = 3
stroke.Transparency = 0
stroke.Parent = button

-- Cores RGB piscantes (com tempo mais lento)
local cores = {
	Color3.fromRGB(0, 170, 255),    -- Azul
	Color3.fromRGB(255, 255, 0),    -- Amarelo
	Color3.fromRGB(255, 0, 0),      -- Vermelho
	Color3.fromRGB(170, 0, 255),    -- Roxo
	Color3.fromRGB(150, 75, 0),     -- Marrom
}
local corIndex = 1

task.spawn(function()
	while true do
		stroke.Color = cores[corIndex]
		corIndex = corIndex % #cores + 1
		task.wait(0.6) -- Mais lento
	end
end)

-- Função para teleportar e deixar o jogador cair
local function teleportar()
	local character = player.Character or player.CharacterAdded:Wait()
	local root = character:FindFirstChild("HumanoidRootPart")

	if root then
		local startPos = root.Position
		local teleportHeight = 500 -- altura segura, não vai pro void
		root.CFrame = CFrame.new(startPos.X, startPos.Y + teleportHeight, startPos.Z)
	end
end

button.MouseButton1Click:Connect(teleportar)
