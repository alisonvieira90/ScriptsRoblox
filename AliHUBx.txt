local Players = game:GetService("Players")
local player = Players.LocalPlayer

-- Criar GUI
local screenGui = Instance.new("ScreenGui")
screenGui.Name = "TpBaseGUI"
screenGui.ResetOnSpawn = false
screenGui.Parent = player:WaitForChild("PlayerGui")

-- Criar botão
local button = Instance.new("TextButton")
button.Name = "TpBase"
button.Size = UDim2.new(0, 120, 0, 40)
button.Position = UDim2.new(0, 10, 1, -50) -- canto inferior esquerdo
button.BackgroundColor3 = Color3.fromRGB(30, 30, 30)
button.TextColor3 = Color3.new(1, 1, 1)
button.Text = "TpBase"
button.Font = Enum.Font.GothamBold
button.TextSize = 16
button.BorderSizePixel = 0
button.BackgroundTransparency = 0.2
button.Parent = screenGui

-- Subida segura
local isMoving = false
local targetY = 44.399

button.MouseButton1Click:Connect(function()
	if isMoving then return end
	local character = player.Character or player.CharacterAdded:Wait()
	local root = character:WaitForChild("HumanoidRootPart")

	if math.abs(root.Position.Y - targetY) < 1 then return end -- já está lá

	isMoving = true

	local bv = Instance.new("BodyVelocity")
	bv.Name = "TpUpVelocity"
	bv.MaxForce = Vector3.new(0, math.huge, 0)
	bv.Velocity = Vector3.new(0, 25, 0)
	bv.Parent = root

	-- Monitorar subida
	local heartbeatConn
	heartbeatConn = game:GetService("RunService").Heartbeat:Connect(function()
		if not root or not root.Parent then
			bv:Destroy()
			heartbeatConn:Disconnect()
			isMoving = false
			return
		end

		if root.Position.Y >= targetY then
			bv:Destroy()
			heartbeatConn:Disconnect()
			isMoving = false
			-- Para garantir que pare suavemente
			root.AssemblyLinearVelocity = Vector3.zero
		end
	end)
end)
